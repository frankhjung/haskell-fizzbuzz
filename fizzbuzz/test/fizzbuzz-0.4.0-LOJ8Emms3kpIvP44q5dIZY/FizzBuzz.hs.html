   15 module FizzBuzz (fizzbuzz1, fizzbuzz2) where
   16 
   17 import           Data.List  (zipWith)
   18 import           Data.Maybe (fromMaybe)
   19 
   20 -- | Given n returns fizzbuzzes.
   21 --
   22 --    * fizz if modulo 3
   23 --    * buzz if modulo 5
   24 --    * fizzbuzz if modulo 3 and 5
   25 --    * n for all other numbers
   26 --
   27 --  Source: https://themonadreader.files.wordpress.com/2014/04/fizzbuzz.pdf
   28 fizzbuzz1 :: Int -> String
   29 fizzbuzz1 n = (fizz . buzz) id (show n)
   30   where
   31     fizz | n `mod` 3 == 0 = \x -> const ("Fizz" ++ x "")
   32          | otherwise = id
   33     buzz | n `mod` 5 == 0 = \x -> const ("Buzz" ++ x "")
   34          | otherwise = id
   35 
   36 -- | Semigroup resonance FizzBuzz by Mark Seemann.
   37 --
   38 -- This "FizzBuzz Kata uses the fundamental concepts of catamorphism, semigroup
   39 -- No if-then-else instructions or pattern matching is required.
   40 -- Instead, we use the string concatenation semigroup to enable resonance between
   41 -- two pulses, and the maybe catamorphism to combine with the list of numbers."
   42 --
   43 -- Source: https://blog.ploeh.dk/2019/12/30/semigroup-resonance-fizzbuzz/
   44 fizzbuzz2 :: Int -> [String]
   45 fizzbuzz2 n = zipWith fromMaybe numbers fizzBuzzes
   46   where
   47     numbers :: [String]
   48     numbers = show <$> [1..n]
   49     fizzes :: [Maybe String]
   50     fizzes = cycle [Nothing, Nothing, Just "Fizz"]
   51     buzzes :: [Maybe String]
   52     buzzes = cycle [Nothing, Nothing, Nothing, Nothing, Just "Buzz"]
   53     fizzBuzzes :: [Maybe String]
   54     fizzBuzzes = zipWith (<>) fizzes buzzes
